截取屏幕指定区域并保存为临时文件。

param region: 截取区域的坐标 (left, top, width, height)

return: 临时文件的路径

def capture_screen_region(region):
    try:
        # 截取屏幕指定区域
        screenshot = pyautogui.screenshot(region=region)
        
        # 创建临时文件
        with tempfile.NamedTemporaryFile(suffix=".png", delete=False) as temp_file:
            temp_path = temp_file.name
            screenshot.save(temp_path)
        
        return temp_path, screenshot  # 返回临时文件路径和截图对象
    except Exception as e:
        print(f"截图失败：{e}")
        return None, None

初始化 PaddleOCR（全局变量，避免重复初始化）

ocr = PaddleOCR(
    use_angle_cls=False,  # 不使用方向分类模型
    lang="ch",           # 使用中文模型
    det_model_dir="ch_PP-OCRv2_det",  # 指定检测模型路径
    rec_model_dir="ch_PP-OCRv2_rec",  # 指定识别模型路径
    show_log=False       # 关闭日志输出
)
print('ocr 完成初始化')

对指定图片进行 OCR 识别，返回识别出的文本

img_path: 图片路径

return: 识别出的文本（字符串）

def ocr_image(img_path):
    try:
        # 进行 OCR 识别
        results = ocr.ocr(img_path, cls=True)
        
        # 提取识别结果
        text_lines = []
        for result in results:
            for line in result:
                text_lines.append(line[1][0])  # 提取每一行的文本
        
        # 将识别结果合并为一个字符串
        text = "\n".join(text_lines)
        return text
    except Exception as e:
        print(f"OCR识别失败：{e}")
        return None

# 从剪贴板获取截屏图片
from PIL import ImageGrab  # 用于从剪贴板获取图片

def get_screenshot_from_clipboard():
    try:
        # 从剪贴板获取图片
        screenshot = ImageGrab.grabclipboard()
        if screenshot is None:
            raise ValueError("剪贴板中没有图片！")
        return screenshot
    except Exception as e:
        print(f"无法从剪贴板获取图片：{e}")
        return None
# 将文本复制到剪贴板
def copy_to_clipboard(text):
    pyperclip.copy(text)
    print("文本已复制到剪贴板！")
